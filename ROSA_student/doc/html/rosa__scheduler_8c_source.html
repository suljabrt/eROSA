29 /***********************************************************
30  * scheduler
31  *
32  * Comment:
33  * Minimalistic scheduler for round robin task switch.
34  * This scheduler choose the next task to execute by looking
35  * at the nexttcb of the current running task.
36  **********************************************************/
37 void scheduler(void)
38 {
39  //Find the next task to execute
40  if (PREEMPTASK == NULL) {
41  PA[EXECTASK->priority] = EXECTASK;
42  EXECTASK = EXECTASK->nexttcb;
43  }
44  else {
45  EXECTASK = PREEMPTASK;
46  PREEMPTASK = NULL;
47  }
48 }

425 int16_t ROSA_delay(uint64_t ticks)
426 {
427  uint64_t dv;
428  uint8_t pr;
429 
430  interruptDisable();
431 
432  dv = ROSA_getTickCount() + ticks;
433  pr = EXECTASK->priority;
434 
435  /* Extract task from its queue */
436  ROSA_TM_ACTION(PA[pr], EXECTASK, Uninstall);
437 
438  EXECTASK->delay = dv;
439 
440  /* Check the current queue for emptiness */
441  if (PA[pr])
442  {
443  PREEMPTASK = PA[pr]->nexttcb;
444  }
445  else
446  {
447  PREEMPTASK = readyQueueSearch();
448  }
449 
450  if (!DQ)
451  {
452  DQ = EXECTASK;
453  EXECTASK->nexttcb = DQ;
454  }
455  else if (DQ->nexttcb == DQ)
456  {
457  DQ->nexttcb = EXECTASK;
458  EXECTASK->nexttcb = DQ;
459  }
460  else
461  {
462  TCBLIST = DQ;
463  while (!( (TCBLIST->delay <= dv) && (dv <= TCBLIST->nexttcb->delay) ) && (TCBLIST->nexttcb != DQ))
464  TCBLIST = TCBLIST->nexttcb;
465  ROSA_tcbInstall(EXECTASK);
466  }
467 
468  if (DQ->delay > EXECTASK->delay)
469  {
470  DQ = EXECTASK;
471  }
472 
473  interruptEnable();
474 
475  ROSA_yield();
476 
477  if (ROSA_getTickCount()>dv)
478  {
479  return dv-ROSA_getTickCount();
480  }
481 
482  return 0;
483 }

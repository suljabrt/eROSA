217 void ROSA_tcbInstall(tcb * tcbTask)
218 {
219  tcb * tcbTmp;
220 
221  /* Is this the first tcb installed? */
222  if(TCBLIST == NULL) {
223  TCBLIST = tcbTask;
224  TCBLIST->nexttcb = tcbTask; //Install the first tcb
225  tcbTask->nexttcb = TCBLIST; //Make the list circular
226  }
227  else {
228  tcbTmp = TCBLIST; //Find last tcb in the list
229  while(tcbTmp->nexttcb != TCBLIST) {
230  tcbTmp = tcbTmp->nexttcb;
231  }
232  tcbTmp->nexttcb = tcbTask; //Install tcb last in the list
233  tcbTask->nexttcb = TCBLIST; //Make the list circular
234  }
235 }

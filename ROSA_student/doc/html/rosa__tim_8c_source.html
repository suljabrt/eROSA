42 /***********************************************************
43  * timerInterruptHandler
44  *
45  * Comment:
46  * This is the basic timer interrupt service routine.
47  **********************************************************/
48 __attribute__((__interrupt__))
49 void timerISR(void)
50 {
51  int sr;
52  volatile avr32_tc_t * tc = &AVR32_TC;
53  systemTick++;
54 
55  //Read the timer status register to determine if this is a valid interrupt
56  sr = tc->channel[0].sr;
57 
58  bool interruptTask = false;
59 
60  while (DELAYQUEUE->delay <= systemTick)
61  {
62  ROSA_taskHandle_t * tmptsk = DELAYQUEUE;
63  removeDelayQueue(&DELAYQUEUE);
64  rqi(&tmptsk);
65  int priority = rqsearch();
66  PREEMPTASK = PA[priority];
67  interruptTask = true;
68  }
69  if (interruptTask)
70  {
71  ROSA_yieldFromISR();
72  }
73 
74 }

92 int16_t ROSA_delay(uint64_t ticks)
93 {
94  rqe(&EXECTASK);

123 /************************************************************************/
124 /* insertDelayQueue() */
125 /* */
126 /* Inserts the given task into the delay queue, before any tasks with */
127 /* a later deadline or lower priority */
128 /************************************************************************/
129 int insertDelayQueue(ROSA_taskHandle_t ** pth, uint64_t deadline)
130 {
131  if (DELAYQUEUE == NULL) {
132  DELAYQUEUE = *pth;
133  DELAYQUEUE->nexttcb = NULL;
134  return 0;
135  }
136  else {
137  ROSA_taskHandle_t * next = DELAYQUEUE;
138  ROSA_taskHandle_t * prev;
139  while (next->delay > (*pth)->delay)
140  {
141  prev = next;
142  next = next->nexttcb;
143  }
144  while (next->priority >= (*pth)->priority && next->delay == (*pth)->delay)
145  {
146  prev = next;
147  next = next->nexttcb;
148  }
149  (*pth)->nexttcb = next;
150  prev->nexttcb = *pth;
151  return 0;
152  }
153 }

155 /************************************************************************/
156 /* removeDelayQueue() */
157 /* */
158 /* Removes the given task from the delay queue */
159 /************************************************************************/
160 int removeDelayQueue(ROSA_taskHandle_t ** pth)
161 {
162  if (*pth == DELAYQUEUE)
163  {
164  DELAYQUEUE = NULL;
165  return 0;
166  }
167  ROSA_taskHandle_t * pt = DELAYQUEUE;
168  while (pt->nexttcb != *pth)
169  {
170  pt = pt->nexttcb;
171  }
172  pt->nexttcb = (*pth)->nexttcb;
173  return 0;
174 }
